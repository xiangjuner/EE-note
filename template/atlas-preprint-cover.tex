\graphicspath{{logos/}}

%-------------------------------------------------------------------------------
% Generic document information
%-------------------------------------------------------------------------------

% Take information from the metadata file you have for your document
% \input{mydocument-metadata}

% or specify all information in this document
% The paper title:
\AtlasTitle{A standalone title page for CERN preprints}

% CERN preprint number -> can be found in CDS entry for Draft 2!
\PreprintIdNumber{CERN-EP-2021-XX}

% Submission journal
\AtlasJournal{\PRL}

% Abstract
\AtlasAbstract{%
  This is a test of how a title page for ATLAS-CERN preprints can be produced
  as a standalone document.
  The idea is that it should be included before the paper is in the journal format.
  With this, there is no need for CDS to stamp the CERN logo and preprint number
  and risk an overlapping with the title and/or other information.
}
